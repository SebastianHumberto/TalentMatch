<div class="canvas-grid">
  <div class="canvas-box canvas-key">
    <h3>Socios clave</h3>
    <ul>
      <li>Universidades e institutos tecnológicos</li>
      <li>Comunidades tech y bootcamps</li>
      <li>Proveedores cloud y pasarelas de pago</li>
      <li>Plataformas profesionales (LinkedIn) para captación indirecta</li>
    </ul>
  </div>

  <div class="canvas-box canvas-activities">
    <h3>Actividades clave</h3>
    <ul>
      <li>Desarrollo y mantenimiento del software</li>
      <li>Mejora continua del algoritmo de matching</li>
      <li>Marketing digital y adquisición de empresas</li>
      <li>Análisis de comportamiento de usuarios</li>
    </ul>
  </div>

  <div class="canvas-box canvas-resources">
    <h3>Recursos clave</h3>
    <ul>
      <li>Plataforma web y base de datos en la nube</li>
      <li>Algoritmo de coincidencia inteligente</li>
      <li>Equipo de desarrollo y marketing</li>
      <li>Infraestructura en AWS o Google Cloud</li>
    </ul>
  </div>

  <div class="canvas-box canvas-value">
    <h3>Propuesta de valor</h3>
    <ul>
      <li>Conectar empresas y talento profesional mediante algoritmos inteligentes</li>
      <li>Reducir tiempo de contratación y mejorar visibilidad profesional</li>
      <li>Recomendaciones y procesos automatizados</li>
    </ul>
  </div>

  <div class="canvas-box canvas-relations">
    <h3>Relaciones con clientes</h3>
    <ul>
      <li>Plataforma autoservicio con panel personalizado</li>
      <li>Recomendaciones automáticas basadas en datos</li>
      <li>Soporte técnico por chat y fidelización para premium</li>
    </ul>
  </div>

  <div class="canvas-box canvas-channels">
    <h3>Canales</h3>
    <ul>
      <li>Plataforma web (canal principal)</li>
      <li>Aplicación móvil (futura)</li>
      <li>Redes sociales y marketing digital</li>
      <li>Email marketing automatizado</li>
    </ul>
  </div>

  <div class="canvas-box canvas-customers">
    <h3>Segmento de clientes</h3>
    <ul>
      <li>B2B: empresas, startups y empresas tecnológicas</li>
      <li>PyMEs en proceso de digitalización</li>
      <li>B2C: profesionales en búsqueda de empleo, técnicos y universitarios</li>
    </ul>
  </div>

  <div class="canvas-box canvas-costs">
    <h3>Estructura de costos</h3>
    <ul>
      <li>Hosting cloud y servidores</li>
      <li>Licencias de software y seguridad</li>
      <li>Desarrollo, marketing y soporte técnico</li>
    </ul>
  </div>

  <div class="canvas-box canvas-revenue">
    <h3>Fuente de ingresos</h3>
    <ul>
      <li>Plan freemium para usuarios</li>
      <li>Suscripción SaaS para empresas</li>
      <li>Publicaciones destacadas de ofertas y servicios de visibilidad</li>
      <li>Pagos integrados: Culqi, Niubiz, PayU</li>
    </ul>
  </div>
